API Reference

Complete reference for all attributes, traits, facades, and commands in the model-agnostic authorization package.

Model-Agnostic: All methods work with any Eloquent model (Article, TeamMember, Organization, etc.). Abilities belong to entities, not users.

PHP 8 Attributes

Declarative authorization using modern PHP attributes on controller methods.

#[HasAny]

Requires the entity to have at least ONE of the specified abilities.

#[HasAny(
    array $actions,              // Required: Abilities to check
    string $model,               // Required: Model class (Article::class, etc.)
    string $modelProperty = 'id' // Optional: Property to lookup entity by
)]

Examples:

use Akindutire\Authorization\Attributes\{HasAny, SubjectValue};
use App\Models\{Article, TeamMember, Organization};

// Check if ARTICLE has ability to be edited
#[HasAny(['can_be_edited'], Article::class, 'id')]
public function update(#[SubjectValue] int $id, Request $request)
{
    // Only executes if the Article has 'can_be_edited' ability
    $article = Article::find($id);
    $article->update($request->validated());
}

// Check if TEAM MEMBER has ability to invite
#[HasAny(['can_invite'], TeamMember::class, 'id')]
public function invite(#[SubjectValue] int $id)
{
    // Only executes if TeamMember has 'can_invite' ability
}

// Lookup by custom property (slug, uuid, etc.)
#[HasAny(['can_be_viewed'], Article::class, 'slug')]
public function show(#[SubjectValue] string $slug)
{
    // Looks up Article by slug, checks if it has 'can_be_viewed'
    return Article::where('slug', $slug)->firstOrFail();
}

#[HasAll]

Requires the entity to have ALL of the specified abilities.

#[HasAll(
    array $actions,              // Required: Abilities to check (must have all)
    string $model,               // Required: Model class
    string $modelProperty = 'id' // Optional: Property to lookup entity by
)]

Examples:

// Article must have BOTH abilities
#[HasAll(['can_be_published', 'can_be_featured'], Article::class, 'id')]
public function publishAndFeature(#[SubjectValue] int $id)
{
    $article = Article::find($id);
    $article->update(['published_at' => now(), 'featured' => true]);
}

// TeamMember must have ALL listed abilities
#[HasAll(['can_invite', 'can_manage_billing'], TeamMember::class, 'id')]
public function adminAction(#[SubjectValue] int $id)
{
    // Only executes if member has both abilities
}

// Organization must have API access
#[HasAll(['can_use_api'], Organization::class, 'org_id')]
public function apiAccess(#[SubjectValue] int $org_id)
{
    // Only executes if organization has API ability
}

#[SubjectValue]

Marks the parameter containing the entity identifier value.

#[SubjectValue(string $parameterName)]  // Extract from Request object
#[SubjectValue]                         // Use parameter directly

Examples:

// Extract from route parameter directly
#[HasAny(['can_be_edited'], Article::class, 'id')]
public function update(#[SubjectValue] int $id, Request $request)
{
    // Uses $id parameter as the lookup value
}

// Extract from Request object
#[HasAny(['can_invite'], TeamMember::class, 'id')]
public function invite(#[SubjectValue('member_id')] Request $request)
{
    // Extracts $request->member_id as the lookup value
}

// Works with any property type
#[HasAny(['can_be_viewed'], Article::class, 'slug')]
public function show(#[SubjectValue] string $slug)
{
    // Uses $slug parameter to find Article by slug column
}

HasPermissions Trait

Add this trait to any Eloquent model to enable ability management.

use Akindutire\Authorization\Traits\HasPermissions;

class Article extends Model
{
    use HasPermissions;

    protected $casts = [
        'allowed_permissions' => 'array',
        'revoked_permissions' => 'array',
    ];
}

grantPermission()

Grant abilities to an entity. Uses atomic operations for race-condition-free updates.

public function grantPermission(string|array $permission): bool

Examples:

// Grant single ability to Article
$article = Article::find(1);
$article->grantPermission('can_be_published');

// Grant multiple abilities to Article
$article->grantPermission(['can_be_edited', 'can_be_published', 'can_be_featured']);

// Grant abilities to TeamMember
$member = TeamMember::find(1);
$member->grantPermission(['can_invite', 'can_manage_billing']);

// Grant abilities to Organization
$org = Organization::find(1);
$org->grantPermission(['can_use_api', 'can_export_data', 'can_white_label']);

revokePermission()

Revoke abilities from an entity. Revoked abilities override granted ones.

public function revokePermission(string|array $permission): bool

Examples:

// Revoke single ability from Article
$article->revokePermission('can_be_deleted');

// Revoke multiple abilities
$article->revokePermission(['can_be_deleted', 'can_be_featured']);

// Lock article from editing
$article->revokePermission('can_be_edited');

// Prevent TeamMember from managing billing
$member->revokePermission('can_manage_billing');
Revocation Override: Revoked abilities always take precedence. If an entity has both granted and revoked entries for the same ability, the revocation wins.

hasPermission()

Check if entity has a specific ability (respects revocations).

public function hasPermission(string $permission): bool

Examples:

// Check if Article has ability
$article = Article::find(1);
if ($article->hasPermission('can_be_published')) {
    // Article can be published
}

// Check if TeamMember has ability
$member = TeamMember::find(1);
if ($member->hasPermission('can_invite')) {
    // Member can invite others
}

// Revoked abilities return false
$article->grantPermission('can_be_edited');
$article->revokePermission('can_be_edited');
$article->hasPermission('can_be_edited'); // Returns: false

hasAnyPermission()

Check if entity has at least one of the specified abilities.

public function hasAnyPermission(array $permissions): bool

Examples:

// Article needs at least one ability
if ($article->hasAnyPermission(['can_be_edited', 'can_be_published'])) {
    // Article has at least one of these abilities
}

// TeamMember needs at least one admin ability
if ($member->hasAnyPermission(['can_invite', 'can_manage_billing', 'can_remove'])) {
    // Member has at least one admin ability
}

hasAllPermissions()

Check if entity has all of the specified abilities.

public function hasAllPermissions(array $permissions): bool

Examples:

// Article must have both abilities
if ($article->hasAllPermissions(['can_be_published', 'can_be_featured'])) {
    // Article can be published AND featured
}

// TeamMember must have all admin abilities
if ($member->hasAllPermissions(['can_invite', 'can_manage_billing'])) {
    // Member has full admin access
}

getAllowedPermissions()

Get all granted abilities for an entity.

public function getAllowedPermissions(): array

Examples:

$abilities = $article->getAllowedPermissions();
// Returns: ['can_be_edited', 'can_be_published', 'can_be_featured']

$memberAbilities = $member->getAllowedPermissions();
// Returns: ['can_invite', 'can_manage_billing', 'can_view_analytics']

getRevokedPermissions()

Get all revoked abilities for an entity.

public function getRevokedPermissions(): array

Examples:

$revoked = $article->getRevokedPermissions();
// Returns: ['can_be_deleted']

$memberRevoked = $member->getRevokedPermissions();
// Returns: ['can_remove_members']

getEffectivePermissions()

Get effective abilities (granted minus revoked).

public function getEffectivePermissions(): array

Examples:

$article->grantPermission(['can_be_edited', 'can_be_published', 'can_be_deleted']);
$article->revokePermission('can_be_deleted');

$effective = $article->getEffectivePermissions();
// Returns: ['can_be_edited', 'can_be_published']
// (can_be_deleted is excluded because it's revoked)

EntityPermission Facade

Fluent interface for checking entity abilities programmatically.

use Akindutire\Authorization\Facades\EntityPermission;

subject()

Set the entity to check abilities on.

public function subject(Model $subject): PermissionSvc

Examples:

// Check Article abilities
$article = Article::find(1);
EntityPermission::subject($article)->hasAny(['can_be_edited']);

// Check TeamMember abilities
$member = TeamMember::find(1);
EntityPermission::subject($member)->hasAll(['can_invite']);

// Check Organization abilities
$org = Organization::find(1);
EntityPermission::subject($org)->hasAny(['can_use_api']);

hasAny()

Check if subject has any of the specified abilities.

public function hasAny(array $actions): bool

Examples:

$canEdit = EntityPermission::subject($article)
    ->hasAny(['can_be_edited', 'can_be_published']);

if ($canEdit) {
    // Article has at least one ability
}

$canInvite = EntityPermission::subject($member)
    ->hasAny(['can_invite', 'can_manage_team']);

hasAll()

Check if subject has all of the specified abilities.

public function hasAll(array $actions): bool

Examples:

$canPublishAndFeature = EntityPermission::subject($article)
    ->hasAll(['can_be_published', 'can_be_featured']);

if ($canPublishAndFeature) {
    // Article has both abilities
}

$hasFullAccess = EntityPermission::subject($org)
    ->hasAll(['can_use_api', 'can_export_data']);

getAbilities()

Get ability template from config for a specific role.

public function getAbilities(string $role): array

Examples:

// Get template from config
$editorAbilities = EntityPermission::getAbilities('article_editor');
// Returns: ['can_be_edited', 'can_be_deleted']

$adminAbilities = EntityPermission::getAbilities('team_admin');
// Returns: ['can_invite', 'can_remove', 'can_manage_billing']

// Grant template abilities to entities
$article->grantPermission($editorAbilities);
$member->grantPermission($adminAbilities);

Artisan Commands

make:permission-migration

Generate migration to add ability columns to a table.

php artisan make:permission-migration {table}

Examples:

# For articles
php artisan make:permission-migration articles

# For team members
php artisan make:permission-migration team_members

# For organizations
php artisan make:permission-migration organizations

Creates migration with:

  • allowed_permissions JSON column
  • revoked_permissions JSON column
  • Database-specific optimizations (MySQL, PostgreSQL, SQLite)

permission:cache

Pre-warm reflection metadata cache for all routes with attributes.

php artisan permission:cache

Output Example:

Warming permission reflection caches...
✓ Permission cache warmed successfully
  Cached: 45 routes
  Skipped: 12 routes

Benefits:

  • Eliminates first-request reflection overhead (~50μs per route)
  • Improves production performance
  • Recommended in deployment scripts

permission:cache-clear

Clear all permission caches (reflection metadata and entity permissions).

php artisan permission:cache-clear

# Clear only reflection metadata
php artisan permission:cache-clear --reflection

# Clear only entity caches
php artisan permission:cache-clear --entities

Output Example:

Clearing permission caches...
✓ Permission caches cleared successfully
  Reflection cache cleared: 45 keys
  Entity cache cleared: 128 keys
Auto-Invalidation: Entity caches automatically clear when abilities change. Manual clearing is rarely needed except for troubleshooting.

Middleware

ValidateSubjectAction

Automatically validates entity abilities based on controller method attributes.

use Akindutire\Authorization\Middleware\ValidateSubjectAction;

// Register in app/Http/Kernel.php
protected $middlewareAliases = [
    'validate.subject.action' => ValidateSubjectAction::class,
];

// Apply to routes
Route::middleware(['validate.subject.action'])->group(function () {
    Route::put('/articles/{id}', [ArticleController::class, 'update']);
    Route::post('/teams/{id}/invite', [TeamController::class, 'invite']);
});

How It Works:

  1. Parses #[HasAny] and #[HasAll] attributes from controller methods
  2. Caches reflection metadata for performance (<1μs for cached checks)
  3. Extracts entity identifier from #[SubjectValue] parameter
  4. Looks up entity using specified model and property
  5. Checks if entity has required abilities
  6. Returns 403 Forbidden if abilities are missing
  7. Allows request to proceed if authorized
Automatic Validation: No manual permission checking needed. The middleware handles everything based on your attribute declarations.

Performance Characteristics

Operation Performance Notes
Cached reflection lookup <1μs Sub-microsecond for warmed cache
First reflection parse ~50-100μs One-time cost per route (cached after)
Entity permission check ~10-20μs Simple array operations in PHP
Entity lookup (cached) <1ms Redis/Memcached retrieval
Entity lookup (database) ~1-5ms Standard Eloquent query
Total overhead (cached) <1ms Negligible impact on response time

Production Benchmarks (50M entities):

  • Cache hit rate: 99%+
  • Permission check (p50): 8ms
  • Database queries/sec: 50 (99.5% reduction)
  • Throughput: 10k+ req/sec
Production Ready: With proper caching, authorization checks add <1ms overhead per request. Scales to millions of entities with constant-time lookups.

Complete Example

Full controller showing all API features:

use Akindutire\Authorization\Attributes\{HasAny, HasAll, SubjectValue};
use Akindutire\Authorization\Facades\EntityPermission;
use App\Models\Article;

class ArticleController extends Controller
{
    // Check if ARTICLE has ability to be viewed
    #[HasAny(['can_be_viewed'], Article::class, 'slug')]
    public function show(#[SubjectValue] string $slug)
    {
        return Article::where('slug', $slug)->firstOrFail();
    }

    // Check if ARTICLE has ability to be edited
    #[HasAny(['can_be_edited'], Article::class, 'id')]
    public function update(#[SubjectValue] int $id, Request $request)
    {
        $article = Article::find($id);
        $article->update($request->validated());
        return response()->json($article);
    }

    // Article must have BOTH abilities
    #[HasAll(['can_be_published', 'can_be_featured'], Article::class, 'id')]
    public function publishAndFeature(#[SubjectValue] int $id)
    {
        $article = Article::find($id);
        $article->update([
            'published_at' => now(),
            'featured' => true,
        ]);
        return response()->json($article);
    }

    // Programmatic ability check
    public function customLogic(int $id)
    {
        $article = Article::find($id);

        // Using trait methods
        if ($article->hasPermission('can_be_edited')) {
            // Article can be edited
        }

        // Using facade
        $canPublish = EntityPermission::subject($article)
            ->hasAny(['can_be_published']);

        if ($canPublish) {
            // Article can be published
        }

        // Grant abilities
        $article->grantPermission(['can_be_shared']);

        // Revoke abilities
        $article->revokePermission(['can_be_deleted']);

        return response()->json([
            'allowed' => $article->getAllowedPermissions(),
            'revoked' => $article->getRevokedPermissions(),
            'effective' => $article->getEffectivePermissions(),
        ]);
    }
}
