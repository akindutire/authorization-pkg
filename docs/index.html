See It In Action

use Akindutire\Authorization\Attributes\{HasAny, SubjectValue};
use App\Models\User;

class ArticleController extends Controller
{
    #[HasAny(['can_edit', 'can_publish'], User::class, 'id')]
    public function update(
        #[SubjectValue('user_id')] Request $request,
        Article $article
    ) {
        // Only users with 'can_edit' OR 'can_publish' permission can access
        $article->update($request->validated());

        return response()->json($article);
    }
}
use Akindutire\Authorization\Traits\HasPermissions;

class User extends Authenticatable
{
    use HasPermissions;

    // $casts automatically added by trait!
}

// Grant permissions
$user->grantPermission(['can_edit', 'can_publish']);

// Revoke permissions
$user->revokePermission('can_publish');

// Check permissions
if ($user->hasPermission('can_edit')) {
    // User can edit
}
use Akindutire\Authorization\Facades\EntityPermission;

// Check via facade
$canEdit = EntityPermission::subject($user)->hasAny(['can_edit']);

// Check all permissions
$isAdmin = EntityPermission::subject($user)
    ->hasAll(['can_edit', 'can_delete', 'can_publish']);

// Get role-based abilities
$permissions = EntityPermission::getAbilities('owner');
