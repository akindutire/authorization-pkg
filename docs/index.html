See It In Action

use Akindutire\Authorization\Attributes\{HasAny, SubjectValue};
use App\Models\Article;

class ArticleController extends Controller
{
    // Check if ARTICLE has abilities (not user!)
    #[HasAny(['can_be_edited'], Article::class, 'id')]
    public function update(
        #[SubjectValue] int $id,
        Request $request
    ) {
        // Only executes if the Article has 'can_be_edited' ability
        $article = Article::find($id);
        $article->update($request->validated());

        return response()->json($article);
    }
}
use Akindutire\Authorization\Traits\HasPermissions;

// Works with ANY model!
class Article extends Model { use HasPermissions; }
class TeamMember extends Model { use HasPermissions; }
class Organization extends Model { use HasPermissions; }

// Grant abilities to an Article
$article->grantPermission(['can_be_edited', 'can_be_published']);

// TeamMember has role abilities
$member->grantPermission(['can_invite', 'can_manage_team']);

// Check abilities
if ($article->hasPermission('can_be_edited')) {
    // Article can be edited
}
use Akindutire\Authorization\Facades\EntityPermission;

// Check if Article has abilities
$canEdit = EntityPermission::subject($article)->hasAny(['can_be_edited']);

// Check TeamMember abilities
$canInvite = EntityPermission::subject($member)->hasAll(['can_invite']);

// Get role-based ability templates
$abilities = EntityPermission::getAbilities('admin');
